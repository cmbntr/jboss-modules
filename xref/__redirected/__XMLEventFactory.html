50   * A redirected XMLEventFactory

52   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
53   * @authore Jason T. Greene

55  @SuppressWarnings("unchecked")
56  public final class __XMLEventFactory extends XMLEventFactory {
57      private static final Constructor<? extends XMLEventFactory> PLATFORM_FACTORY;
58      private static volatile Constructor<? extends XMLEventFactory> DEFAULT_FACTORY;
59  
60      static {
61          Thread thread = Thread.currentThread();
62          ClassLoader old = thread.getContextClassLoader();
63  
64          // Unfortunately we can not use null because of a stupid bug in the jdk JAXP factory finder.
65          // Lack of tccl causes the provider file discovery to fallback to the jaxp loader (bootclasspath)
66          // which is correct. However, after parsing it, it then disables the fallback for the loading of the class.
67          // Thus, the class can not be found.
68          //
69          // Work around the problem by using the System CL, although in the future we may want to just "inherit"
70          // the environment's TCCL
71          thread.setContextClassLoader(ClassLoader.getSystemClassLoader());
72          try {
73              XMLEventFactory factory = XMLEventFactory.newInstance();
74              try {
75                  DEFAULT_FACTORY = PLATFORM_FACTORY = factory.getClass().getConstructor();
76              } catch (NoSuchMethodException e) {
77                  throw __RedirectedUtils.wrapped(new NoSuchMethodError(e.getMessage()), e);
78              }
79              System.setProperty(XMLEventFactory.class.getName(), __XMLEventFactory.class.getName());
80          } finally {
81              thread.setContextClassLoader(old);
82          }
83      }
84  
85      public static void changeDefaultFactory(ModuleIdentifier id, ModuleLoader loader) {
86          Class<? extends XMLEventFactory> clazz = __RedirectedUtils.loadProvider(id, XMLEventFactory.class, loader);
87          if (clazz != null) {
88              try {
89                  DEFAULT_FACTORY = clazz.getConstructor();
90              } catch (NoSuchMethodException e) {
91                  throw __RedirectedUtils.wrapped(new NoSuchMethodError(e.getMessage()), e);
92              }
93          }
94      }
95  
96      public static void restorePlatformFactory() {
97          DEFAULT_FACTORY = PLATFORM_FACTORY;
98      }
