23  package org.jboss.modules;
24  
25  import org.jboss.modules.filter.PathFilter;
26  import org.jboss.modules.filter.PathFilters;
27  
28  /**
29   * A specification of a resource loader within a module.  A resource loader may optionally be associated with a
30   * path filter which can be used to decide which paths of a resource loader to include.
31   *
32   * @apiviz.exclude
33   *
34   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
35   */
36  public final class ResourceLoaderSpec {
37      private final ResourceLoader resourceLoader;
38      private final PathFilter pathFilter;
39  
40      ResourceLoaderSpec(final ResourceLoader resourceLoader, final PathFilter pathFilter) {
41          this.resourceLoader = resourceLoader;
42          this.pathFilter = pathFilter;
43      }
44  
45      /**
46       * Construct a new instance.
47       *
48       * @param resourceLoader the resource loader to include
49       * @param pathFilter the path filter to apply to the resource loader's paths
50       * @return the specification
51       */
52      public static ResourceLoaderSpec createResourceLoaderSpec(final ResourceLoader resourceLoader, final PathFilter pathFilter) {
53          return new ResourceLoaderSpec(resourceLoader, pathFilter);
54      }
55  
56      /**
57       * Construct a new instance which accepts all paths in the resource loader.
58       *
59       * @param resourceLoader the resource loader to include
60       * @return the specification
61       */
62      public static ResourceLoaderSpec createResourceLoaderSpec(final ResourceLoader resourceLoader) {
63          return new ResourceLoaderSpec(resourceLoader, PathFilters.acceptAll());
64      }
65  
66      ResourceLoader getResourceLoader() {
67          return resourceLoader;
68      }
69  
70      PathFilter getPathFilter() {
71          return pathFilter;
72      }
73  }
