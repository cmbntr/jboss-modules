23  package org.jboss.modules;
24  
25  import java.io.Closeable;
26  import java.io.File;
27  import java.io.IOException;
28  import java.io.InputStream;
29  import java.util.jar.Attributes;
30  import java.util.jar.JarEntry;
31  import java.util.jar.JarFile;
32  import java.util.jar.Manifest;
33  import org.jboss.modules.filter.MultiplePathFilterBuilder;
34  import org.jboss.modules.filter.PathFilters;

36  /**
37   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
38   */
39  final class JarModuleLoader extends ModuleLoader {
40  
41      private static final String[] NO_STRINGS = new String[0];
42      private final ModuleLoader delegate;
43      private final JarFile jarFile;
44      private final String simpleJarName;
45      private final ModuleIdentifier myIdentifier;
46  
47      JarModuleLoader(final ModuleLoader delegate, final JarFile jarFile) {
48          this.delegate = delegate;
49          this.jarFile = jarFile;
50          String jarName = jarFile.getName();
51          simpleJarName = jarName.substring(jarName.lastIndexOf(File.separatorChar) + 1);
52          myIdentifier = ModuleIdentifier.create(simpleJarName);
53      }
54  
55      protected Module preloadModule(final ModuleIdentifier identifier) throws ModuleLoadException {
56          if (identifier.equals(myIdentifier)) {
57              return loadModuleLocal(identifier);
58          } else {
59              Module module = loadModuleLocal(identifier);
60              if (module == null) {
61                  return preloadModule(identifier, delegate);
62              } else {
63                  return module;
64              }
65          }
66      }
67  
68      ModuleIdentifier getMyIdentifier() {
69          return myIdentifier;
70      }
71  
72      public String toString() {
73          return "JAR module loader";
74      }
75  
76      protected ModuleSpec findModule(final ModuleIdentifier identifier) throws ModuleLoadException {
77          if (identifier.equals(myIdentifier)) {
78              // special root JAR module
79              Manifest manifest;
80              try {
81                  manifest = jarFile.getManifest();
82              } catch (IOException e) {
83                  throw new ModuleLoadException("Failed to load MANIFEST from JAR", e);
84              }
85              ModuleSpec.Builder builder = ModuleSpec.build(identifier);
86              Attributes mainAttributes = manifest.getMainAttributes();
87              String mainClass = mainAttributes.getValue(Attributes.Name.MAIN_CLASS);
88              if (mainClass != null) {
89                  builder.setMainClass(mainClass);
90              }
91              String classPath = mainAttributes.getValue(Attributes.Name.CLASS_PATH);
92              String dependencies = mainAttributes.getValue("Dependencies");
93              MultiplePathFilterBuilder pathFilterBuilder = PathFilters.multiplePathFilterBuilder(true);
94              pathFilterBuilder.addFilter(PathFilters.is("modules"), false);
95              pathFilterBuilder.addFilter(PathFilters.isChildOf("modules"), false);
96              builder.addResourceRoot(ResourceLoaderSpec.createResourceLoaderSpec(new JarFileResourceLoader("", jarFile), pathFilterBuilder.create()));
97              String[] classPathEntries = classPath == null ? NO_STRINGS : classPath.split("\\w+");
98              for (String entry : classPathEntries) {
99                  if (! entry.isEmpty()) {
100                     if (entry.startsWith("../") || entry.startsWith("./") || entry.startsWith("/") || entry.contains("/../")) {
101                         // invalid
102                         continue;
103                     }
104                     if (entry.endsWith("/")) {
105                         // directory reference
106                         File root = new File(jarFile.getName(), entry);
107                         FileResourceLoader resourceLoader = new FileResourceLoader(entry, root);
108                         builder.addResourceRoot(ResourceLoaderSpec.createResourceLoaderSpec(resourceLoader));
109                     } else {
110                         // assume a JAR
111                         File root = new File(jarFile.getName(), entry);
112                         JarFile childJarFile;
113                         try {
114                             childJarFile = new JarFile(root, true);
115                         } catch (IOException e) {
116                             // ignore and continue
117                             continue;
118                         }
119                         builder.addResourceRoot(ResourceLoaderSpec.createResourceLoaderSpec(new JarFileResourceLoader(entry, childJarFile)));
120                     }
121                 }
122             }
123             String[] dependencyEntries = dependencies == null ? NO_STRINGS : dependencies.split("\\s*,\\s*");
124             for (String dependencyEntry : dependencyEntries) {
125                 boolean optional = false;
126                 boolean export = false;
127                 dependencyEntry = dependencyEntry.trim();
128                 if (! dependencyEntry.isEmpty()) {
129                     String[] fields = dependencyEntry.split("\\s+");
130                     if (fields.length < 1) {
131                         continue;
132                     }
133                     String moduleName = fields[0];
134                     for (int i = 1; i < fields.length; i++) {
135                         String field = fields[i];
136                         if (field.equals("optional")) {
137                             optional = true;
138                         } else if (field.equals("export")) {
139                             export = true;
140                         }
141                         // else ignored
142                     }
143                     builder.addDependency(DependencySpec.createModuleDependencySpec(ModuleIdentifier.fromString(moduleName), export, optional));
144                 }
145             }
146             builder.addDependency(DependencySpec.createSystemDependencySpec(JDKPaths.JDK));
147             builder.addDependency(DependencySpec.createLocalDependencySpec());
148             return builder.create();
149         } else {
150             String namePath = identifier.getName().replace('.', '/');
151             String basePath = "modules/" + namePath + "/" + identifier.getSlot();
152             JarEntry moduleXmlEntry = jarFile.getJarEntry(basePath + "/module.xml");
153             if (moduleXmlEntry == null) {
154                 return null;
155             }
156             ModuleSpec moduleSpec;
157             try {
158                 InputStream inputStream = jarFile.getInputStream(moduleXmlEntry);
159                 try {
160                     moduleSpec = ModuleXmlParser.parseModuleXml(new ModuleXmlParser.ResourceRootFactory() {
161                         public ResourceLoader createResourceLoader(final String rootPath, final String loaderPath, final String loaderName) throws IOException {
162                             return new JarFileResourceLoader(loaderName, jarFile, loaderPath);
163                         }
164                     }, basePath, inputStream, moduleXmlEntry.getName(), identifier);
165                 } finally {
166                     safeClose(inputStream);
167                 }
168             } catch (IOException e) {
169                 throw new ModuleLoadException("Failed to read module.xml file", e);
170             }
171             return moduleSpec;
172         }
173     }
174 
175     private static void safeClose(final Closeable closeable) {
176         if (closeable != null) try {
177             closeable.close();
178         } catch (IOException e) {
179             // ignore
180         }
181     }
