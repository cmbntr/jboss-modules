94      /**
95       * Run JBoss Modules.
96       *
97       * @param args the command-line arguments
98       *
99       * @throws Throwable if an error occurs
100      */
101     public static void main(String[] args) throws Throwable {
102         final int argsLen = args.length;
103         String deps = null;
104         String[] moduleArgs = NO_STRINGS;
105         String modulePath = null;
106         String classpath = null;
107         boolean jar = false;
108         boolean classpathDefined = false;
109         boolean classDefined = false;
110         String moduleIdentifierOrExeName = null;
111         ModuleIdentifier jaxpModuleIdentifier = null;
112         boolean defaultSecMgr = false;
113         String secMgrModule = null;
114         for (int i = 0, argsLength = argsLen; i < argsLength; i++) {
115             final String arg = args[i];
116             try {
117                 if (arg.charAt(0) == '-') {
118                     // it's an option
119                     if ("-version".equals(arg)) {
120                         System.out.println("JBoss Modules version " + getVersionString());
121                         return;
122                     } else if ("-help".equals(arg)) {
123                         usage();
124                         return;
125                     } else if ("-modulepath".equals(arg) || "-mp".equals(arg)) {
126                         if (modulePath != null) {
127                             System.err.println("Module path may only be specified once");
128                             System.exit(1);
129                         }
130                         modulePath = args[++i];
131                         System.setProperty("module.path", modulePath);
132                     } else if ("-config".equals(arg)) {
133                         System.err.println("Config files are no longer supported.  Use the -mp option instead");
134                         System.exit(1);
135                     } else if ("-jaxpmodule".equals(arg)) {
136                         jaxpModuleIdentifier = ModuleIdentifier.fromString(args[++i]);
137                     } else if ("-jar".equals(arg)) {
138                         if (jar) {
139                             System.err.println("-jar flag may only be specified once");
140                             System.exit(1);
141                         }
142                         if (classpathDefined) {
143                             System.err.println("-cp/-classpath may not be specified with -jar");
144                             System.exit(1);
145                         }
146                         if (classDefined) {
147                             System.err.println("-class may not be specified with -jar");
148                             System.exit(1);
149                         }
150                         jar = true;
151                     } else if ("-cp".equals(arg) || "-classpath".equals(arg)) {
152                         if (classpathDefined) {
153                             System.err.println("-cp or -classpath may only be specified once.");
154                             System.exit(1);
155                         }
156                         if (classDefined) {
157                             System.err.println("-class may not be specified with -cp/classpath");
158                             System.exit(1);
159                         }
160                         if (jar) {
161                             System.err.println("-cp/-classpath may not be specified with -jar");
162                             System.exit(1);
163                         }
164                         classpathDefined = true;
165                         classpath = args[++i];
166                         AccessController.doPrivileged(new PropertyWriteAction("java.class.path", classpath));
167                     } else if ("-dep".equals(arg) || "-dependencies".equals(arg)) {
168                         if (deps != null) {
169                             System.err.println("-dep or -dependencies may only be specified once.");
170                             System.exit(1);
171                         }
172                         deps = args[++i];
173                     } else if ("-class".equals(arg)) {
174                         if (classDefined) {
175                             System.err.println("-class flag may only be specified once");
176                             System.exit(1);
177                         }
178                         if (classpathDefined) {
179                             System.err.println("-class may not be specified with -cp/classpath");
180                             System.exit(1);
181                         }
182                         if (jar) {
183                             System.err.println("-class may not be specified with -jar");
184                             System.exit(1);
185                         }
186                         classDefined = true;
187                     } else if ("-logmodule".equals(arg)) {
188                         System.err.println("WARNING: -logmodule is deprecated. Please use the system property 'java.util.logging.manager' or the 'java.util.logging.LogManager' service loader.");
189                         i++;
190                     } else if ("-secmgr".equals(arg)) {
191                         if (defaultSecMgr) {
192                             System.err.println("-secmgr may only be specified once");
193                             System.exit(1);
194                         }
195                         if (secMgrModule != null) {
196                             System.err.println("-secmgr may not be specified when -secmgrmodule is given");
197                             System.exit(1);
198                         }
199                         defaultSecMgr = true;
200                     } else if ("-secmgrmodule".equals(arg)) {
201                         if (secMgrModule != null) {
202                             System.err.println("-secmgrmodule may only be specified once");
203                             System.exit(1);
204                         }
205                         if (defaultSecMgr) {
206                             System.err.println("-secmgrmodule may not be specified when -secmgr is given");
207                             System.exit(1);
208                         }
209                         secMgrModule = args[++i];
210                     } else {
211                         System.err.printf("Invalid option '%s'\n", arg);
212                         usage();
213                         System.exit(1);
214                     }
215                 } else {
216                     // it's the module specification
217                     moduleIdentifierOrExeName = arg;
218                     int cnt = argsLen - i - 1;
219                     moduleArgs = new String[cnt];
220                     System.arraycopy(args, i + 1, moduleArgs, 0, cnt);
221                     break;
222                 }
223             } catch (IndexOutOfBoundsException e) {
224                 System.err.printf("Argument expected for option %s\n", arg);
225                 usage();
226                 System.exit(1);
227             }
228         }
229 
230         if (deps != null && ! classDefined && ! classpathDefined) {
231             System.err.println("-deps may only be specified when -cp/-classpath or -class is in use");
232             System.exit(1);
233         }
234 
235         // run the module
236         if (moduleIdentifierOrExeName == null) {
237             if (classDefined || classpathDefined) {
238                 System.err.println("No class name specified");
239             } else if (jar) {
240                 System.err.println("No JAR specified");
241             } else {
242                 System.err.println("No module specified");
243             }
244             usage();
245             System.exit(1);
246         }
247         final ModuleLoader loader;
248         final ModuleLoader environmentLoader;
249         environmentLoader = DefaultBootModuleLoaderHolder.INSTANCE;
250         final ModuleIdentifier moduleIdentifier;
251         if (jar) {
252             loader = new JarModuleLoader(environmentLoader, new JarFile(moduleIdentifierOrExeName));
253             moduleIdentifier = ((JarModuleLoader) loader).getMyIdentifier();
254         } else if (classpathDefined || classDefined) {
255             loader = new ClassPathModuleLoader(environmentLoader, moduleIdentifierOrExeName, classpath, deps);
256             moduleIdentifier = ModuleIdentifier.CLASSPATH;
257         } else {
258             loader = environmentLoader;
259             moduleIdentifier = ModuleIdentifier.fromString(moduleIdentifierOrExeName);
260         }
261         Module.initBootModuleLoader(loader);
262         if (jaxpModuleIdentifier != null) {
263             __JAXPRedirected.changeAll(jaxpModuleIdentifier, Module.getBootModuleLoader());
264         }
265 
266         if (secMgrModule != null) {
267             final Module loadedModule;
268             try {
269                 loadedModule = loader.loadModule(ModuleIdentifier.fromString(secMgrModule));
270             } catch (ModuleNotFoundException e) {
271                 e.printStackTrace(System.err);
272                 System.exit(1);
273                 return;
274             }
275             final Iterator<SecurityManager> iterator = ServiceLoader.load(SecurityManager.class, loadedModule.getClassLoaderPrivate()).iterator();
276             if (iterator.hasNext()) {
277                 System.setSecurityManager(iterator.next());
278             } else {
279                 System.err.println("No security manager found in module " + secMgrModule);
280                 System.exit(1);
281             }
282         }
283 
284         final Module module;
285         try {
286             module = loader.loadModule(moduleIdentifier);
287         } catch (ModuleNotFoundException e) {
288             e.printStackTrace(System.err);
289             System.exit(1);
290             return;
291         }
292 
293         ModularURLStreamHandlerFactory.addHandlerModule(module);
294         ModularContentHandlerFactory.addHandlerModule(module);
295 
296         if (defaultSecMgr) {
297             final Iterator<SecurityManager> iterator = ServiceLoader.load(SecurityManager.class, module.getClassLoaderPrivate()).iterator();
298             if (iterator.hasNext()) {
299                 System.setSecurityManager(iterator.next());
300             } else {
301                 System.setSecurityManager(new SecurityManager());
302             }
303         }
304 
305         final ModuleClassLoader bootClassLoader = module.getClassLoaderPrivate();
306         setContextClassLoader(bootClassLoader);
307 
308         final String logManagerName = getServiceName(bootClassLoader, "java.util.logging.LogManager");
309         if (logManagerName != null) {
310             System.setProperty("java.util.logging.manager", logManagerName);
311             if (LogManager.getLogManager().getClass() == LogManager.class) {
312                 System.err.println("WARNING: Failed to load the specified log manager class " + logManagerName);
313             } else {
314                 Module.setModuleLogger(new JDKModuleLogger());
315             }
316         }
317 
318         final String mbeanServerBuilderName = getServiceName(bootClassLoader, "javax.management.MBeanServerBuilder");
319         if (mbeanServerBuilderName != null) {
320             System.setProperty("javax.management.builder.initial", mbeanServerBuilderName);
321             // Initialize the platform mbean server
322             ManagementFactory.getPlatformMBeanServer();
323         }
324 
325         try {
326             ModuleLoader.installMBeanServer();
327             module.run(moduleArgs);
328         } catch (InvocationTargetException e) {
329             throw e.getCause();
330         }
331         return;
332     }

334     private static String getServiceName(ClassLoader classLoader, String className) throws IOException {
335         final InputStream stream = classLoader.getResourceAsStream("META-INF/services/" + className);
336         if (stream != null) {
337             try {
338                 final BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
339                 String line;
340                 while ((line = reader.readLine()) != null) {
341                     final int i = line.indexOf('#');
342                     if (i != -1) {
343                         line = line.substring(0, i);
344                     }
345                     line = line.trim();
346                     if (line.length() == 0) continue;
347                     return line;
348                 }
349 
350             } finally {
351                 try {
352                     stream.close();
353                 } catch (IOException ignored) {
354                     // ignore
355                 }
356             }
357         }
358         return null;
359     }

362     private static final String VERSION_STRING;

364     static {
365         final Enumeration<URL> resources;
366         String jarName = "(unknown)";
367         String versionString = "(unknown)";
368         try {
369             final ClassLoader classLoader = Main.class.getClassLoader();
370             resources = classLoader == null ? ModuleClassLoader.getSystemResources("META-INF/MANIFEST.MF") : classLoader.getResources("META-INF/MANIFEST.MF");
371             while (resources.hasMoreElements()) {
372                 final URL url = resources.nextElement();
373                 final InputStream stream = url.openStream();
374                 if (stream != null) try {
375                     final Manifest manifest = new Manifest(stream);
376                     final Attributes mainAttributes = manifest.getMainAttributes();
377                     if (mainAttributes != null && "JBoss Modules".equals(mainAttributes.getValue("Specification-Title"))) {
378                         jarName = mainAttributes.getValue("Jar-Name");
379                         versionString = mainAttributes.getValue("Jar-Version");
380                     }
381                 } finally {
382                     try { stream.close(); } catch (Throwable ignored) {}
383                 }
384             }
385         } catch (IOException ignored) {
386         }
387         JAR_NAME = jarName;
388         VERSION_STRING = versionString;
389     }
