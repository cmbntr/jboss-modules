25  /**
26   * General helpful path utility methods.
27   *
28   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
29   */
30  public final class PathUtils {
31  
32      private PathUtils() {
33      }
34  
35      /**
36       * Canonicalize the given path.  Removes all {@code .} and {@code ..} segments from the path.
37       *
38       * @param path the relative or absolute possibly non-canonical path
39       * @return the canonical path
40       */
41      public static String canonicalize(String path) {
42          final int length = path.length();
43          // 0 - start
44          // 1 - got one .
45          // 2 - got two .
46          // 3 - got /
47          int state = 0;
48          if (length == 0) {
49              return path;
50          }
51          final char[] targetBuf = new char[length];
52          // string segment end exclusive
53          int e = length;
54          // string cursor position
55          int i = length;
56          // buffer cursor position
57          int a = length - 1;
58          // number of segments to skip
59          int skip = 0;
60          loop: while (--i >= 0) {
61              char c = path.charAt(i);
62              outer: switch (c) {
63                  case '/': {
64                      inner: switch (state) {
65                          case 0: state = 3; e = i; break outer;
66                          case 1: state = 3; e = i; break outer;
67                          case 2: state = 3; e = i; skip ++; break outer;
68                          case 3: e = i; break outer;
69                          default: throw new IllegalStateException();
70                      }
71                      // not reached!
72                  }
73                  case '.': {
74                      inner: switch (state) {
75                          case 0: state = 1; break outer;
76                          case 1: state = 2; break outer;
77                          case 2: break inner; // emit!
78                          case 3: state = 1; break outer;
79                          default: throw new IllegalStateException();
80                      }
81                      // fall thru
82                  }
83                  default: {
84                      final int newE = e > 0 ? path.lastIndexOf('/', e - 1) : -1;
85                      final int segmentLength = e - newE - 1;
86                      if (skip > 0) {
87                          skip--;
88                      } else {
89                          if (state == 3) {
90                              targetBuf[a--] = '/';
91                          }
92                          path.getChars(newE + 1, e, targetBuf, (a -= segmentLength) + 1);
93                      }
94                      state = 0;
95                      i = newE + 1;
96                      e = newE;
97                      break;
98                  }
99              }
100         }
101         if (state == 3) {
102             targetBuf[a--] = '/';
103         }
104         return new String(targetBuf, a + 1, length - a - 1);
105     }
