23  package org.jboss.modules;
24  
25  import java.security.PrivilegedAction;
26  
27  import static java.lang.System.getSecurityManager;
28  import static java.lang.Thread.currentThread;
29  import static java.security.AccessController.doPrivileged;
30  
31  /**
32   * This class must not be public.
33   *
34   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
35   */
36  final class SecurityActions {
37  
38      private static final PrivilegedAction<ClassLoader> GET_LOADER_ACTION = new PrivilegedAction<ClassLoader>() {
39          public ClassLoader run() {
40              return currentThread().getContextClassLoader();
41          }
42      };
43  
44      static ClassLoader setContextClassLoader(final ClassLoader classLoader) {
45          final SecurityManager sm = getSecurityManager();
46          if (sm != null) {
47              return doPrivileged(new PrivilegedAction<ClassLoader>() {
48                  public ClassLoader run() {
49                      try {
50                          return currentThread().getContextClassLoader();
51                      } finally {
52                          currentThread().setContextClassLoader(classLoader);
53                      }
54                  }
55              });
56          } else {
57              try {
58                  return currentThread().getContextClassLoader();
59              } finally {
60                  currentThread().setContextClassLoader(classLoader);
61              }
62          }
63      }
64  
65      static ClassLoader getContextClassLoader() {
66          final SecurityManager sm = getSecurityManager();
67          if (sm != null) {
68              return doPrivileged(GET_LOADER_ACTION);
69          } else {
70              return currentThread().getContextClassLoader();
71          }
72      }
73  }
