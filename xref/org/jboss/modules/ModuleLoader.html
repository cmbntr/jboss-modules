245     /**
246      * Try to load a module from this module loader.  Returns {@code null} if the module is not found.  The returned
247      * module may not yet be resolved.  The returned module may have a different name than the given identifier if
248      * the identifier is an alias for another module.
249      *
250      * @param identifier the module identifier
251      * @return the module
252      * @throws ModuleLoadException if an error occurs while loading the module
253      */
254     protected final Module loadModuleLocal(ModuleIdentifier identifier) throws ModuleLoadException {
255         FutureModule futureModule = moduleMap.get(identifier);
256         if (futureModule != null) {
257             return futureModule.getModule();
258         }
259 
260         FutureModule newFuture = new FutureModule(identifier);
261         futureModule = moduleMap.putIfAbsent(identifier, newFuture);
262         if (futureModule != null) {
263             return futureModule.getModule();
264         }
265 
266         boolean ok = false;
267         try {
268             final ModuleLogger log = Module.log;
269             log.trace("Locally loading module %s from %s", identifier, this);
270             final long startTime = Metrics.getCurrentCPUTime();
271             final ModuleSpec moduleSpec = findModule(identifier);
272             loadTimeUpdater.addAndGet(this, Metrics.getCurrentCPUTime() - startTime);
273             if (moduleSpec == null) {
274                 log.trace("Module %s not found from %s", identifier, this);
275                 return null;
276             }
277             if (! moduleSpec.getModuleIdentifier().equals(identifier)) {
278                 throw new ModuleLoadException("Module loader found a module with the wrong name");
279             }
280             final Module module;
281             if ( moduleSpec instanceof AliasModuleSpec) {
282                 final ModuleIdentifier aliasTarget = ((AliasModuleSpec) moduleSpec).getAliasTarget();
283                 try {
284                     newFuture.setModule(module = loadModuleLocal(aliasTarget));
285                 } catch (RuntimeException e) {
286                     log.trace(e, "Failed to load module %s (alias for %s)", identifier, aliasTarget);
287                     throw e;
288                 } catch (Error e) {
289                     log.trace(e, "Failed to load module %s (alias for %s)", identifier, aliasTarget);
290                     throw e;
291                 }
292             } else {
293                 module = defineModule((ConcreteModuleSpec) moduleSpec, newFuture);
294             }
295             log.trace("Loaded module %s from %s", identifier, this);
296             ok = true;
297             return module;
298         } finally {
299             if (! ok) {
300                 newFuture.setModule(null);
301                 moduleMap.remove(identifier, newFuture);
302             }
303         }
304     }

306     /**
307      * Find an already-loaded module, returning {@code null} if the module isn't currently loaded.  May block
308      * while the loaded state of the module is in question (if the module is being concurrently loaded from another
309      * thread, for example).
310      *
311      * @param identifier the module identifier
312      * @return the module, or {@code null} if it wasn't found
313      */
314     protected final Module findLoadedModuleLocal(ModuleIdentifier identifier) {
315         FutureModule futureModule = moduleMap.get(identifier);
316         if (futureModule != null) {
317             try {
318                 return futureModule.getModule();
319             } catch (ModuleNotFoundException e) {
320                 return null;
321             }
322         } else {
323             return null;
324         }
325     }

327     /**
328      * Unload a module from this module loader.  Note that this has no effect on existing modules which refer to the
329      * module being unloaded.  Also, only modules from the current module loader can be unloaded.  Unloading the same
330      * module more than once has no additional effect.  This method only removes the mapping for the module; any running
331      * threads which are currently accessing or linked to the module will continue to function, however attempts to load
332      * this module will fail until a new module is loaded with the same name.  Once this happens, if all references to
333      * the previous module are not cleared, the same module may be loaded more than once, causing possible class duplication
334      * and class cast exceptions if proper care is not taken.
335      *
336      * @param module the module to unload
337      * @throws SecurityException if an attempt is made to unload a module which does not belong to this module loader
338      */
339     protected final void unloadModuleLocal(Module module) throws SecurityException {
340         final ModuleLoader moduleLoader = module.getModuleLoader();
341         if (moduleLoader != this) {
342             throw new SecurityException("Attempted to unload " + module + " from a different module loader");
343         }
344         final ModuleIdentifier id = module.getIdentifier();
345         final FutureModule futureModule = moduleMap.get(id);
346         if (futureModule.module == module) {
347             moduleMap.remove(id, futureModule);
348         }
349     }

365     /**
366      * Defines a Module based on a specification.  May only be called from {@link #loadModuleLocal(ModuleIdentifier)}.
367      *
368      * @param moduleSpec The module specification to create the Module from
369      * @param futureModule the future module to populate
370      * @return The defined Module
371      * @throws ModuleLoadException If any dependent modules can not be loaded
372      */
373     private Module defineModule(final ConcreteModuleSpec moduleSpec, final FutureModule futureModule) throws ModuleLoadException {
374 
375         final ModuleLogger log = Module.log;
376         final ModuleIdentifier moduleIdentifier = moduleSpec.getModuleIdentifier();
377 
378         final Module module = new Module(moduleSpec, this);
379         module.getClassLoaderPrivate().recalculate();
380         module.setDependencies(Arrays.asList(moduleSpec.getDependencies()));
381         log.moduleDefined(moduleIdentifier, this);
382         try {
383             futureModule.setModule(module);
384             return module;
385         } catch (RuntimeException e) {
386             log.trace(e, "Failed to load module %s", moduleIdentifier);
387             throw e;
388         } catch (Error e) {
389             log.trace(e, "Failed to load module %s", moduleIdentifier);
390             throw e;
391         }
392     }

484     private static final class FutureModule {
485         private static final Object NOT_FOUND = new Object();
486 
487         private final ModuleIdentifier identifier;
488         private volatile Object module;
489 
490         FutureModule(final ModuleIdentifier identifier) {
491             this.identifier = identifier;
492         }
493 
494         Module getModule() throws ModuleNotFoundException {
495             boolean intr = false;
496             try {
497                 Object module = this.module;
498                 if (module == null) synchronized (this) {
499                     while ((module = this.module) == null) {
500                         try {
501                             wait();
502                         } catch (InterruptedException e) {
503                             intr = true;
504                         }
505                     }
506                 }
507                 if (module == NOT_FOUND) throw new ModuleNotFoundException(identifier.toString());
508                 return (Module) module;
509             } finally {
510                 if (intr) Thread.currentThread().interrupt();
511             }
512         }
513 
514         void setModule(Module m) {
515             synchronized (this) {
516                 module = m == null ? NOT_FOUND : m;
517                 notifyAll();
518             }
519         }
520     }

592         private void doDumpModuleInformation(final String name, final StringBuilder b) {
593             ModuleInfo description = getModuleDescription(name);
594             b.append("Module ").append(name).append('\n');
595             b.append("    Class loader: ").append(description.getClassLoader()).append('\n');
596             String fallbackLoader = description.getFallbackLoader();
597             if (fallbackLoader != null) b.append("    Fallback loader: ").append(fallbackLoader).append('\n');
598             String mainClass = description.getMainClass();
599             if (mainClass != null) b.append("    Main Class: ").append(mainClass).append('\n');
600             List<ResourceLoaderInfo> loaders = description.getResourceLoaders();
601             b.append("    Resource Loaders:\n");
602             for (ResourceLoaderInfo loader : loaders) {
603                 b.append("        Loader Type: ").append(loader.getType()).append('\n');
604                 b.append("        Paths:\n");
605                 for (String path : loader.getPaths()) {
606                     b.append("            ").append(path).append('\n');
607                 }
608             }
609             b.append("    Dependencies:\n");
610             for (DependencyInfo dependencyInfo : description.getDependencies()) {
611                 b.append("        Type: ").append(dependencyInfo.getDependencyType()).append('\n');
612                 String moduleName = dependencyInfo.getModuleName();
613                 if (moduleName != null) {
614                     b.append("        Module Name: ").append(moduleName).append('\n');
615                 }
616                 if (dependencyInfo.isOptional()) b.append("        (optional)\n");
617                 b.append("        Export Filter: ").append(dependencyInfo.getExportFilter()).append('\n');
618                 b.append("        Import Filter: ").append(dependencyInfo.getImportFilter()).append('\n');
619                 String localLoader = dependencyInfo.getLocalLoader();
620                 if (localLoader != null) {
621                     b.append("        Local Loader: ").append(localLoader).append('\n');
622                     b.append("        Paths:\n");
623                     for (String path : dependencyInfo.getLocalLoaderPaths()) {
624                         b.append("            ").append(path).append('\n');
625                     }
626                 }
627             }
628         }

630         public boolean unloadModule(final String name) {
631             final SecurityManager sm = System.getSecurityManager();
632             if (sm != null) {
633                 sm.checkPermission(MODULE_UNLOAD_ANY_PERM);
634             }
635             final ModuleLoader loader = getModuleLoader();
636             final Module module = loader.findLoadedModuleLocal(ModuleIdentifier.fromString(name));
637             if (module == null) {
638                 return false;
639             } else {
640                 loader.unloadModuleLocal(module);
641                 return true;
642             }
643         }
