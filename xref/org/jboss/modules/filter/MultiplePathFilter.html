23  package org.jboss.modules.filter;
24  
25  import java.util.Arrays;
26  
27  /**
28   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
29   */
30  final class MultiplePathFilter implements PathFilter {
31      private final PathFilter[] filters;
32      private final boolean[] includeFlag;
33      private final boolean defaultVal;
34      private final int hashCode;
35  
36      MultiplePathFilter(final PathFilter[] filters, final boolean[] includeFlag, final boolean defaultVal) {
37          this.filters = filters;
38          this.includeFlag = includeFlag;
39          this.defaultVal = defaultVal;
40          hashCode = Boolean.valueOf(defaultVal).hashCode() * 13 + (Arrays.hashCode(includeFlag) * 13 + (Arrays.hashCode(filters)));
41      }
42  
43      public boolean accept(final String path) {
44          final int len = filters.length;
45          for (int i = 0; i < len; i++) {
46              if (filters[i].accept(path)) return includeFlag[i];
47          }
48          return defaultVal;
49      }
50  
51      public String toString() {
52          StringBuilder builder = new StringBuilder();
53          builder.append("multi-path filter {");
54          int len = filters.length;
55          for (int i = 0; i < len; i++) {
56              final PathFilter filter = filters[i];
57              final boolean include = includeFlag[i];
58              builder.append(include ? "include " : "exclude ");
59              builder.append(filter);
60              builder.append(", ");
61          }
62          builder.append("default ").append(defaultVal ? "accept" : "reject");
63          builder.append('}');
64          return builder.toString();
65      }
66  
67      public int hashCode() {
68          return hashCode;
69      }
70  
71      public boolean equals(Object other) {
72          return other instanceof MultiplePathFilter && equals((MultiplePathFilter)other);
73      }
74  
75      public boolean equals(MultiplePathFilter other) {
76          return this == other || other != null && Arrays.equals(filters, other.filters) && Arrays.equals(includeFlag, other.includeFlag) && defaultVal == other.defaultVal;
77      }
78  }
