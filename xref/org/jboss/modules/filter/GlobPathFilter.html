23  package org.jboss.modules.filter;
24  
25  import java.util.regex.Matcher;
26  import java.util.regex.Pattern;
27  
28  /**
29   * Default implementation of PathFilter.  Uses glob based includes and excludes to determine whether to export.  
30   *
31   * @author John E. Bailey
32   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
33   */
34  final class GlobPathFilter implements PathFilter {
35      private static final Pattern GLOB_PATTERN = Pattern.compile("(\\*\\*?)|(\\?)|(\\\\.)|(/+)|([^*?]+)");
36  
37      private final String glob;
38      private final Pattern pattern;
39  
40      /**
41       * Construct a new instance.
42       *
43       * @param glob the path glob to match
44       */
45      GlobPathFilter(final String glob) {
46          pattern = getGlobPattern(glob);
47          this.glob = glob;
48      }
49  
50      /**
51       * Determine whether a path should be accepted.
52       *
53       * @param path the path to check
54       * @return true if the path should be accepted, false if not
55       */
56      public boolean accept(final String path) {
57          return pattern.matcher(path).matches();
58      }
59  
60      /**
61       * Get a regular expression pattern which accept any path names which match the given glob.  The glob patterns
62       * function similarly to {@code ant} file patterns.  Valid metacharacters in the glob pattern include:
63       * <ul>
64       * <li><code>"\"</code> - escape the next character (treat it literally, even if it is itself a recognized metacharacter)</li>
65       * <li><code>"?"</code> - match any non-slash character</li>
66       * <li><code>"*"</code> - match zero or more non-slash characters</li>
67       * <li><code>"**"</code> - match zero or more characters, including slashes</li>
68       * <li><code>"/"</code> - match one or more slash characters.  Consecutive {@code /} characters are collapsed down into one.</li>
69       * </ul>
70       * In addition, any glob pattern matches all subdirectories thereof.  A glob pattern ending in {@code /} is equivalent
71       * to a glob pattern ending in <code>/**</code> in that the named directory is not itself included in the glob.
72       * <p/>
73       * <b>See also:</b> <a href="http://ant.apache.org/manual/dirtasks.html#patterns">"Patterns" in the Ant Manual</a>
74       *
75       * @param glob the glob to match
76       *
77       * @return the pattern
78       */
79      private static Pattern getGlobPattern(final String glob) {
80          StringBuilder patternBuilder = new StringBuilder();
81          final Matcher m = GLOB_PATTERN.matcher(glob);
82          boolean lastWasSlash = false;
83          while (m.find()) {
84              lastWasSlash = false;
85              String grp;
86              if ((grp = m.group(1)) != null) {
87                  // match a * or **
88                  if (grp.length() == 2) {
89                      // it's a **
90                      patternBuilder.append(".*");
91                  } else {
92                      // it's a *
93                      patternBuilder.append("[^/]*");
94                  }
95              } else if ((grp = m.group(2)) != null) {
96                  // match a '?' glob pattern; any non-slash character
97                  patternBuilder.append("[^/]");
98              } else if ((grp = m.group(3)) != null) {
99                  // backslash-escaped value
100                 patternBuilder.append(Pattern.quote(m.group().substring(1)));
101             } else if ((grp = m.group(4)) != null) {
102                 // match any number of / chars
103                 patternBuilder.append("/+");
104                 lastWasSlash = true;
105             } else {
106                 // some other string
107                 patternBuilder.append(Pattern.quote(m.group()));
108             }
109         }
110         if (lastWasSlash) {
111             // ends in /, append **
112             patternBuilder.append(".*");
113         } else {
114             patternBuilder.append("(?:/.*)?");
115         }
116         return Pattern.compile(patternBuilder.toString());
117     }
118 
119     public int hashCode() {
120         return glob.hashCode() + 13;
121     }
122 
123     public boolean equals(final Object obj) {
124         return obj instanceof GlobPathFilter && equals((GlobPathFilter) obj);
125     }
126 
127     public boolean equals(final GlobPathFilter obj) {
128         return obj != null && obj.pattern.equals(pattern);
129     }
130 
131     public String toString() {
132         final StringBuilder b = new StringBuilder();
133         b.append("match ");
134         if (glob != null) {
135             b.append('"').append(glob).append('"');
136         } else {
137             b.append('/').append(pattern).append('/');
138         }
139         return b.toString();
140     }
