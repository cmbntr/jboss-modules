23  package org.jboss.modules;
24  
25  import java.io.File;
26  import java.io.IOException;
27  import java.security.AccessController;
28  import java.util.Enumeration;
29  import java.util.Set;
30  import java.util.zip.ZipEntry;
31  import java.util.zip.ZipFile;
32  
33  /**
34   * A utility class which maintains the set of JDK paths.  Makes certain assumptions about the disposition of the
35   * class loader used to load JBoss Modules; thus this class should only be used when booted up via the "-jar" or "-cp"
36   * switches.
37   *
38   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
39   */
40  final class JDKPaths {
41      static final Set<String> JDK;
42  
43      static {
44          final Set<String> pathSet = new FastCopyHashSet<String>(1024);
45          final Set<String> jarSet = new FastCopyHashSet<String>(1024);
46          final String sunBootClassPath = AccessController.doPrivileged(new PropertyReadAction("sun.boot.class.path"));
47          final String javaClassPath = AccessController.doPrivileged(new PropertyReadAction("java.class.path"));
48          processClassPathItem(sunBootClassPath, jarSet, pathSet);
49          processClassPathItem(javaClassPath, jarSet, pathSet);
50          pathSet.add("org/jboss/modules");
51          pathSet.add("org/jboss/modules/filter");
52          pathSet.add("org/jboss/modules/log");
53          pathSet.add("org/jboss/modules/management");
54          pathSet.add("org/jboss/modules/ref");
55          JDK = pathSet;
56      }
57  
58      private JDKPaths() {
59      }
60  
61      private static void processClassPathItem(final String classPath, final Set<String> jarSet, final Set<String> pathSet) {
62          if (classPath == null) return;
63          int s = 0, e;
64          do {
65              e = classPath.indexOf(File.pathSeparatorChar, s);
66              String item = e == -1 ? classPath.substring(s) : classPath.substring(s, e);
67              if (! jarSet.contains(item)) {
68                  final File file = new File(item);
69                  if (file.isDirectory()) {
70                      processDirectory0(pathSet, file);
71                  } else {
72                      try {
73                          final ZipFile zipFile = new ZipFile(file);
74                          try {
75                              final Enumeration<? extends ZipEntry> entries = zipFile.entries();
76                              while (entries.hasMoreElements()) {
77                                  final ZipEntry entry = entries.nextElement();
78                                  final String name = entry.getName();
79                                  final int lastSlash = name.lastIndexOf('/');
80                                  if (lastSlash != -1) {
81                                      pathSet.add(name.substring(0, lastSlash));
82                                  }
83                              }
84                          } finally {
85                              zipFile.close();
86                          }
87                      } catch (IOException ex) {
88                          // ignore
89                      }
90                  }
91              }
92              s = e + 1;
93          } while (e != -1);
94      }
95  
96      private static void processDirectory0(final Set<String> pathSet, final File file) {
97          for (File entry : file.listFiles()) {
98              if (entry.isDirectory()) {
99                  processDirectory1(pathSet, entry, file.getPath());
100             } else {
101                 final String parent = entry.getParent();
102                 if (parent != null) pathSet.add(parent);
103             }
104         }
105     }
106 
107     private static void processDirectory1(final Set<String> pathSet, final File file, final String pathBase) {
108         for (File entry : file.listFiles()) {
109             if (entry.isDirectory()) {
110                 processDirectory1(pathSet, entry, pathBase);
111             } else {
112                 String packagePath = entry.getParent();
113                 if (packagePath != null) {
114                     packagePath = packagePath.substring(pathBase.length()).replace('\\', '/');;
115                     if(packagePath.startsWith("/")) {
116                         packagePath = packagePath.substring(1);
117                     }
118                     pathSet.add(packagePath);
119                 }
120             }
121         }
122     }
