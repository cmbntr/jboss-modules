87          private static final Map<String, Namespace> namespaces;

89          static {
90              Map<String, Namespace> namespacesMap = new HashMap<String, Namespace>();
91              namespacesMap.put("urn:jboss:module:1.0", MODULE_1_0);
92              namespacesMap.put("urn:jboss:module:1.1", MODULE_1_1);
93              namespacesMap.put("urn:jboss:module:1.2", MODULE_1_2);
94              namespaces = namespacesMap;

98              Namespace namespace = namespaces.get(qName.getNamespaceURI());
99              return namespace == null ? UNKNOWN : namespace;

236     static ModuleSpec parseModuleXml(final ModuleIdentifier moduleIdentifier, final File root, final File moduleInfoFile) throws ModuleLoadException {
237         final FileInputStream fis;
238         try {
239             fis = new FileInputStream(moduleInfoFile);
240         } catch (FileNotFoundException e) {
241             throw new ModuleLoadException("No module.xml file found at " + moduleInfoFile);
242         }
243         try {
244             return parseModuleXml(new ResourceRootFactory() {
245                     public ResourceLoader createResourceLoader(final String rootPath, final String loaderPath, final String loaderName) throws IOException {
246                         File file = new File(rootPath, loaderPath);
247                         if (file.isDirectory()) {
248                             return new FileResourceLoader(loaderName, file);
249                         } else {
250                             final JarFile jarFile = new JarFile(file, true);
251                             return new JarFileResourceLoader(loaderName, jarFile);
252                         }
253                     }
254                 }, root.getPath(), new BufferedInputStream(fis), moduleInfoFile.getPath(), moduleIdentifier);
255         } finally {
256             safeClose(fis);
257         }
258     }

266     private static final XMLInputFactory INPUT_FACTORY = XMLInputFactory.newInstance();

268     static ModuleSpec parseModuleXml(final ResourceRootFactory factory, final String rootPath, InputStream source, final String moduleInfoFile, final ModuleIdentifier moduleIdentifier) throws ModuleLoadException {
269         try {
270             final XMLInputFactory inputFactory = INPUT_FACTORY;
271             setIfSupported(inputFactory, XMLInputFactory.IS_VALIDATING, Boolean.FALSE);
272             setIfSupported(inputFactory, XMLInputFactory.SUPPORT_DTD, Boolean.FALSE);
273             final XMLStreamReader streamReader = inputFactory.createXMLStreamReader(source);
274             try {
275                 return parseDocument(factory, rootPath, streamReader, moduleIdentifier);
276             } finally {
277                 safeClose(streamReader);
278             }
279         } catch (XMLStreamException e) {
280             throw new ModuleLoadException("Error loading module from " + moduleInfoFile, e);
281         }
282     }

300     private static XMLStreamException unexpectedContent(final XMLStreamReader reader) {
301         final String kind;
302         switch (reader.getEventType()) {
303             case ATTRIBUTE: kind = "attribute"; break;
304             case CDATA: kind = "cdata"; break;
305             case CHARACTERS: kind = "characters"; break;
306             case COMMENT: kind = "comment"; break;
307             case DTD: kind = "dtd"; break;
308             case END_DOCUMENT: kind = "document end"; break;
309             case END_ELEMENT: kind = "element end"; break;
310             case ENTITY_DECLARATION: kind = "entity declaration"; break;
311             case ENTITY_REFERENCE: kind = "entity ref"; break;
312             case NAMESPACE: kind = "namespace"; break;
313             case NOTATION_DECLARATION: kind = "notation declaration"; break;
314             case PROCESSING_INSTRUCTION: kind = "processing instruction"; break;
315             case SPACE: kind = "whitespace"; break;
316             case START_DOCUMENT: kind = "document start"; break;
317             case START_ELEMENT: kind = "element start"; break;
318             default: kind = "unknown"; break;
319         }
320         final StringBuilder b = new StringBuilder("Unexpected content of type '").append(kind).append('\'');
321         if (reader.hasName()) {
322             b.append(" named '").append(reader.getName()).append('\'');
323         }
324         if (reader.hasText()) {
325             b.append(", text is: '").append(reader.getText()).append('\'');
326         }
327         return new XMLStreamException(b.toString(), reader.getLocation());
328     }

338     private static XMLStreamException missingAttributes(final Location location, final Set<Attribute> required) {
339         final StringBuilder b = new StringBuilder("Missing one or more required attributes:");
340         for (Attribute attribute : required) {
341             b.append(' ').append(attribute);
342         }
343         return new XMLStreamException(b.toString(), location);
344     }

346     private static ModuleSpec parseDocument(final ResourceRootFactory factory, final String rootPath, XMLStreamReader reader, final ModuleIdentifier moduleIdentifier) throws XMLStreamException {
347         while (reader.hasNext()) {
348             switch (reader.nextTag()) {
349                 case START_DOCUMENT: {
350                     return parseRootElement(factory, rootPath, reader, moduleIdentifier);
351                 }
352                 case START_ELEMENT: {
353                     final Element element = Element.of(reader.getName());
354                     switch (element) {
355                         case MODULE: {
356                             final ModuleSpec.Builder specBuilder = ModuleSpec.build(moduleIdentifier);
357                             parseModuleContents(factory, rootPath, reader, specBuilder);
358                             parseEndDocument(reader);
359                             return specBuilder.create();
360                         }
361                         case MODULE_ALIAS: {
362                             final ModuleSpec moduleSpec = parseModuleAliasContents(reader, moduleIdentifier);
363                             parseEndDocument(reader);
364                             return moduleSpec;
365                         }
366                         case MODULE_ABSENT: {
367                             parseModuleAbsentContents(reader, moduleIdentifier);
368                             return null;
369                         }
370                         default: {
371                             throw unexpectedContent(reader);
372                         }
373                     }
374                 }
375                 default: {
376                     throw unexpectedContent(reader);
377                 }
378             }
379         }
380         throw endOfDocument(reader.getLocation());
381     }

383     private static ModuleSpec parseRootElement(final ResourceRootFactory factory, final String rootPath, final XMLStreamReader reader, final ModuleIdentifier moduleIdentifier) throws XMLStreamException {
384         while (reader.hasNext()) {
385             switch (reader.nextTag()) {
386                 case START_ELEMENT: {
387                     final Element element = Element.of(reader.getName());
388                     switch (element) {
389                         case MODULE: {
390                             final ModuleSpec.Builder specBuilder = ModuleSpec.build(moduleIdentifier);
391                             parseModuleContents(factory, rootPath, reader, specBuilder);
392                             parseEndDocument(reader);
393                             return specBuilder.create();
394                         }
395                         case MODULE_ALIAS: {
396                             final ModuleSpec moduleSpec = parseModuleAliasContents(reader, moduleIdentifier);
397                             parseEndDocument(reader);
398                             return moduleSpec;
399                         }
400                         case MODULE_ABSENT: {
401                             parseModuleAbsentContents(reader, moduleIdentifier);
402                             return null;
403                         }
404                         default: {
405                             throw unexpectedContent(reader);
406                         }
407                     }
408                 }
409                 default: {
410                     throw unexpectedContent(reader);
411                 }
412             }
413         }
414         throw endOfDocument(reader.getLocation());
415     }

417     private static ModuleSpec parseModuleAliasContents(final XMLStreamReader reader, final ModuleIdentifier moduleIdentifier) throws XMLStreamException {
418         final int count = reader.getAttributeCount();
419         String name = null;
420         String slot = null;
421         String targetName = null;
422         String targetSlot = null;
423         final Set<Attribute> required = EnumSet.of(Attribute.NAME, Attribute.TARGET_NAME);
424         for (int i = 0; i < count; i ++) {
425             final Attribute attribute = Attribute.of(reader.getAttributeName(i));
426             required.remove(attribute);
427             switch (attribute) {
428                 case NAME:    name = reader.getAttributeValue(i); break;
429                 case SLOT:    slot = reader.getAttributeValue(i); break;
430                 case TARGET_NAME: targetName = reader.getAttributeValue(i); break;
431                 case TARGET_SLOT: targetSlot = reader.getAttributeValue(i); break;
432                 default: throw unexpectedContent(reader);
433             }
434         }
435         if (! required.isEmpty()) {
436             throw missingAttributes(reader.getLocation(), required);
437         }
438         if (! moduleIdentifier.equals(ModuleIdentifier.create(name, slot))) {
439             throw invalidModuleName(reader.getLocation(), moduleIdentifier);
440         }
441         while (reader.hasNext()) {
442             switch (reader.nextTag()) {
443                 case END_ELEMENT: {
444                     return ModuleSpec.buildAlias(moduleIdentifier, ModuleIdentifier.create(targetName, targetSlot)).create();
445                 }
446                 default: {
447                     throw unexpectedContent(reader);
448                 }
449             }
450         }
451         throw endOfDocument(reader.getLocation());
452     }

455         final int count = reader.getAttributeCount();
456         String name = null;
457         String slot = null;
458         final Set<Attribute> required = EnumSet.of(Attribute.NAME, Attribute.TARGET_NAME);
459         for (int i = 0; i < count; i ++) {
460             final Attribute attribute = Attribute.of(reader.getAttributeName(i));
461             required.remove(attribute);
462             switch (attribute) {
463                 case NAME:    name = reader.getAttributeValue(i); break;
464                 case SLOT:    slot = reader.getAttributeValue(i); break;
465                 default: throw unexpectedContent(reader);
466             }
467         }
468         if (! required.isEmpty()) {
469             throw missingAttributes(reader.getLocation(), required);
470         }
471         if (! moduleIdentifier.equals(ModuleIdentifier.create(name, slot))) {
472             throw invalidModuleName(reader.getLocation(), moduleIdentifier);
473         }
474         while (reader.hasNext()) {
475             switch (reader.nextTag()) {
476                 case END_ELEMENT: {
477                     return;
478                 }
479                 default: {
480                     throw unexpectedContent(reader);
481                 }
482             }
483         }
484         throw endOfDocument(reader.getLocation());

487     private static void parseModuleContents(final ResourceRootFactory factory, final String rootPath, final XMLStreamReader reader, final ModuleSpec.Builder specBuilder) throws XMLStreamException {
488         final int count = reader.getAttributeCount();
489         String name = null;
490         String slot = null;
491         final Set<Attribute> required = EnumSet.of(Attribute.NAME);
492         for (int i = 0; i < count; i ++) {
493             final Attribute attribute = Attribute.of(reader.getAttributeName(i));
494             required.remove(attribute);
495             switch (attribute) {
496                 case NAME:    name = reader.getAttributeValue(i); break;
497                 case SLOT:    slot = reader.getAttributeValue(i); break;
498                 default: throw unexpectedContent(reader);
499             }
500         }
501         if (! required.isEmpty()) {
502             throw missingAttributes(reader.getLocation(), required);
503         }
504         if (! specBuilder.getIdentifier().equals(ModuleIdentifier.create(name, slot))) {
505             throw invalidModuleName(reader.getLocation(), specBuilder.getIdentifier());
506         }
507         // xsd:all
508         MultiplePathFilterBuilder exportsBuilder = PathFilters.multiplePathFilterBuilder(true);
509         Set<Element> visited = EnumSet.noneOf(Element.class);
510         while (reader.hasNext()) {
511             switch (reader.nextTag()) {
512                 case END_ELEMENT: {
513                     specBuilder.addDependency(DependencySpec.createLocalDependencySpec(PathFilters.acceptAll(), exportsBuilder.create()));
514                     return;
515                 }
516                 case START_ELEMENT: {
517                     final Element element = Element.of(reader.getName());
518                     if (visited.contains(element)) {
519                         throw unexpectedContent(reader);
520                     }
521                     visited.add(element);
522                     switch (element) {
523                         case EXPORTS:      parseFilterList(reader, exportsBuilder); break;
524                         case DEPENDENCIES: parseDependencies(reader, specBuilder); break;
525                         case MAIN_CLASS:   parseMainClass(reader, specBuilder); break;
526                         case RESOURCES:    parseResources(factory, rootPath, reader, specBuilder); break;
527                         case PROPERTIES:   parseProperties(reader, specBuilder); break;
528                         default: throw unexpectedContent(reader);
529                     }
530                     break;
531                 }
532                 default: {
533                     throw unexpectedContent(reader);
534                 }
535             }
536         }
537         throw endOfDocument(reader.getLocation());
538     }

540     private static void parseDependencies(final XMLStreamReader reader, final ModuleSpec.Builder specBuilder) throws XMLStreamException {
541         // xsd:choice
542         while (reader.hasNext()) {
543             switch (reader.nextTag()) {
544                 case END_ELEMENT: {
545                     return;
546                 }
547                 case START_ELEMENT: {
548                     switch (Element.of(reader.getName())) {
549                         case MODULE: parseModuleDependency(reader, specBuilder); break;
550                         case SYSTEM: parseSystemDependency(reader, specBuilder); break;
551                         default: throw unexpectedContent(reader);
552                     }
553                     break;
554                 }
555                 default: {
556                     throw unexpectedContent(reader);
557                 }
558             }
559         }
560         throw endOfDocument(reader.getLocation());
561     }

563     private static void parseModuleDependency(final XMLStreamReader reader, final ModuleSpec.Builder specBuilder) throws XMLStreamException {
564         String name = null;
565         String slot = null;
566         boolean export = false;
567         boolean optional = false;
568         Disposition services = Disposition.NONE;
569         final Set<Attribute> required = EnumSet.of(Attribute.NAME);
570         final int count = reader.getAttributeCount();
571         for (int i = 0; i < count; i ++) {
572             final Attribute attribute = Attribute.of(reader.getAttributeName(i));
573             required.remove(attribute);
574             switch (attribute) {
575                 case NAME:    name = reader.getAttributeValue(i); break;
576                 case SLOT:    slot = reader.getAttributeValue(i); break;
577                 case EXPORT:  export = Boolean.parseBoolean(reader.getAttributeValue(i)); break;
578                 case SERVICES:services = Disposition.of(reader.getAttributeValue(i)); break;
579                 case OPTIONAL:optional = Boolean.parseBoolean(reader.getAttributeValue(i)); break;
580                 default: throw unexpectedContent(reader);
581             }
582         }
583         if (! required.isEmpty()) {
584             throw missingAttributes(reader.getLocation(), required);
585         }
586         final MultiplePathFilterBuilder importBuilder = PathFilters.multiplePathFilterBuilder(true);
587         final MultiplePathFilterBuilder exportBuilder = PathFilters.multiplePathFilterBuilder(export);
588         while (reader.hasNext()) {
589             switch (reader.nextTag()) {
590                 case END_ELEMENT: {
591                     if (services == ModuleXmlParser.Disposition.EXPORT) {
592                         // If services are to be re-exported, add META-INF/services -> true near the end of the list
593                         exportBuilder.addFilter(PathFilters.getMetaInfServicesFilter(), true);
594                     }
595                     if (export) {
596                         // If re-exported, add META-INF/** -> false at the end of the list (require explicit override)
597                         exportBuilder.addFilter(PathFilters.getMetaInfSubdirectoriesFilter(), false);
598                         exportBuilder.addFilter(PathFilters.getMetaInfFilter(), false);
599                     }
600                     final PathFilter exportFilter = exportBuilder.create();
601                     final PathFilter importFilter;
602                     if (importBuilder.isEmpty()) {
603                         importFilter = services == Disposition.NONE ? PathFilters.getDefaultImportFilter() : PathFilters.getDefaultImportFilterWithServices();
604                     } else {
605                         if (services != Disposition.NONE) {
606                             importBuilder.addFilter(PathFilters.getMetaInfServicesFilter(), true);
607                         }
608                         importBuilder.addFilter(PathFilters.getMetaInfSubdirectoriesFilter(), false);
609                         importBuilder.addFilter(PathFilters.getMetaInfFilter(), false);
610                         importFilter = importBuilder.create();
611                     }
612                     specBuilder.addDependency(DependencySpec.createModuleDependencySpec(importFilter, exportFilter, null, ModuleIdentifier.create(name, slot), optional));
613                     return;
614                 }
615                 case START_ELEMENT: {
616                     switch (Element.of(reader.getName())) {
617                         case EXPORTS: parseFilterList(reader, exportBuilder); break;
618                         case IMPORTS: parseFilterList(reader, importBuilder); break;
619                         default: throw unexpectedContent(reader);
620                     }
621                     break;
622                 }
623                 default: {
624                     throw unexpectedContent(reader);
625                 }
626             }
627         }
628     }

630     private static void parseSystemDependency(final XMLStreamReader reader, final ModuleSpec.Builder specBuilder) throws XMLStreamException {
631         boolean export = false;
632         final int count = reader.getAttributeCount();
633         for (int i = 0; i < count; i ++) {
634             final Attribute attribute = Attribute.of(reader.getAttributeName(i));
635             switch (attribute) {
636                 case EXPORT:  export = Boolean.parseBoolean(reader.getAttributeValue(i)); break;
637                 default: throw unexpectedContent(reader);
638             }
639         }
640         Set<String> paths = Collections.emptySet();
641         final MultiplePathFilterBuilder exportBuilder = PathFilters.multiplePathFilterBuilder(export);
642         while (reader.hasNext()) {
643             switch (reader.nextTag()) {
644                 case END_ELEMENT: {
645                     final PathFilter exportFilter = exportBuilder.create();
646                     specBuilder.addDependency(DependencySpec.createSystemDependencySpec(PathFilters.acceptAll(), exportFilter, paths));
647                     return;
648                 }
649                 case START_ELEMENT: {
650                     switch (Element.of(reader.getName())) {
651                         case PATHS: {
652                             paths = parseSet(reader);
653                             break;
654                         }
655                         case EXPORTS: {
656                             parseFilterList(reader, exportBuilder);
657                             break;
658                         }
659                         default: {
660                             throw unexpectedContent(reader);
661                         }
662                     }
663                 }
664             }
665         }
666     }

668     private static void parseMainClass(final XMLStreamReader reader, final ModuleSpec.Builder specBuilder) throws XMLStreamException {
669         String name = null;
