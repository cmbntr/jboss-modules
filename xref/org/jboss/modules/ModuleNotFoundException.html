23  package org.jboss.modules;
24  
25  /**
26   * Module not found exceptions are thrown when no module loaders can locate a module which fulfills a given module
27   * identifier.
28   *
29   * @author <a href="mailto:jbailey@redhat.com">John Bailey</a>
30   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
31   */
32  public class ModuleNotFoundException extends ModuleLoadException {
33      private static final long serialVersionUID = -1225396191255481860L;
34  
35      /**
36       * Constructs a {@code ModuleNotFoundException} with no detail message. The cause is not initialized, and may
37       * subsequently be initialized by a call to {@link #initCause(Throwable) initCause}.
38       */
39      public ModuleNotFoundException() {
40      }
41  
42      /**
43       * Constructs a {@code ModuleNotFoundException} with the specified detail message. The cause is not initialized, and may
44       * subsequently be initialized by a call to {@link #initCause(Throwable) initCause}.
45       *
46       * @param msg the detail message
47       */
48      public ModuleNotFoundException(final String msg) {
49          super(msg);
50      }
51  
52      /**
53       * Constructs a {@code ModuleNotFoundException} with the specified cause. The detail message is set to:
54       * <pre>(cause == null ? null : cause.toString())</pre>
55       * (which typically contains the class and detail message of {@code cause}).
56       *
57       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method)
58       */
59      public ModuleNotFoundException(final Throwable cause) {
60          super(cause);
61      }
62  
63      /**
64       * Constructs a {@code ModuleNotFoundException} with the specified detail message and cause.
65       *
66       * @param msg   the detail message
67       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method)
68       */
69      public ModuleNotFoundException(final String msg, final Throwable cause) {
70          super(msg, cause);
71      }
72  }
