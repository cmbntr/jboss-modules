188     /**
189      * Construct a new instance from a module specification.
190      *
191      * @param spec the module specification
192      * @param moduleLoader the module loader
193      */
194     Module(final ConcreteModuleSpec spec, final ModuleLoader moduleLoader) {
195         this.moduleLoader = moduleLoader;
196 
197         // Initialize state from the spec.
198         identifier = spec.getModuleIdentifier();
199         mainClassName = spec.getMainClass();
200         fallbackLoader = spec.getFallbackLoader();
201         //noinspection ThisEscapedInObjectConstruction
202         final ModuleClassLoader.Configuration configuration = new ModuleClassLoader.Configuration(this, spec.getAssertionSetting(), spec.getResourceLoaders(), spec.getClassFileTransformer());
203         final ModuleClassLoaderFactory factory = spec.getModuleClassLoaderFactory();
204         ModuleClassLoader moduleClassLoader = null;
205         if (factory != null) moduleClassLoader = factory.create(configuration);
206         if (moduleClassLoader == null) moduleClassLoader = new ModuleClassLoader(configuration);
207         this.moduleClassLoader = moduleClassLoader;
208         final Map<String, String> properties = spec.getProperties();
209         this.properties = properties.isEmpty() ? Collections.<String, String>emptyMap() : new LinkedHashMap<String, String>(properties);
210     }

235     /**
236      * Run a module's main class, if any.
237      *
238      * @param args the arguments to pass
239      * @throws NoSuchMethodException if there is no main method
240      * @throws InvocationTargetException if the main method failed
241      * @throws ClassNotFoundException if the main class is not found
242      */
243     public void run(final String[] args) throws NoSuchMethodException, InvocationTargetException, ClassNotFoundException {
244         try {
245             if (mainClassName == null) {
246                 throw new NoSuchMethodException("No main class defined for " + this);
247             }
248             final ClassLoader oldClassLoader = SecurityActions.setContextClassLoader(moduleClassLoader);
249             try {
250                 final Class<?> mainClass = Class.forName(mainClassName, false, moduleClassLoader);
251                 try {
252                     Class.forName(mainClassName, true, moduleClassLoader);
253                 } catch (Throwable t) {
254                     throw new InvocationTargetException(t, "Failed to initialize main class '" + mainClassName + "'");
255                 }
256                 final Method mainMethod = mainClass.getMethod("main", String[].class);
257                 final int modifiers = mainMethod.getModifiers();
258                 if (! Modifier.isStatic(modifiers)) {
259                     throw new NoSuchMethodException("Main method is not static for " + this);
260                 }
261                 // ignore the return value
262                 mainMethod.invoke(null, new Object[] {args});
263             } finally {
264                 SecurityActions.setContextClassLoader(oldClassLoader);
265             }
266         } catch (IllegalAccessException e) {
267             // unexpected; should be public
268             throw new IllegalAccessError(e.getMessage());
269         }
270     }

356      * Get the module for a class loader, or {@code null} if the class loader is not associated with any module.  If
357      * the class loader is unknown, it is possible to check the parent class loader up the chain, and so on until a module is found.

364         if (cl instanceof ModuleClassLoader) {
365             return ((ModuleClassLoader) cl).getModule();

368         } else {
369             return null;

374      * Gets the boot module loader. The boot module loader is the
375      * initial loader that is established by the module framework. It typically
376      * is based off of the environmental module path unless it is overridden by
377      * specifying a different class name for the {@code boot.module.loader} system
378      * property.
379      *
380      * @return the boot module loader
381      */
382     public static ModuleLoader getBootModuleLoader() {
383         final SecurityManager sm = System.getSecurityManager();
384         if (sm != null) {
385             sm.checkPermission(GET_BOOT_MODULE_LOADER);
386         }
387         ModuleLoader loader;
388         while ((loader = BOOT_MODULE_LOADER.get()) == null) {
389             loader = DefaultBootModuleLoaderHolder.INSTANCE;
390             if (BOOT_MODULE_LOADER.compareAndSet(null, loader)) {
391                 break;
392             }
393             // get it again
394         }
395         return loader;
396     }

495     /**
496      * Load a class from a local loader.
497      *
498      * @param className the class name
499      * @param resolve {@code true} to resolve the class after definition
500      * @return the class
501      */
502     Class<?> loadModuleClass(final String className, final boolean resolve) {
503         for (String s : systemPackages) {
504             if (className.startsWith(s)) {
505                 try {
506                     return moduleClassLoader.loadClass(className, resolve);
507                 } catch (ClassNotFoundException e) {
508                     return null;
509                 }
510             }
511         }
512         final String path = pathOfClass(className);
513         final Map<String, List<LocalLoader>> paths = getPathsUnchecked();
514         final List<LocalLoader> loaders = paths.get(path);
515         if (loaders != null) {
516             Class<?> clazz;
517             for (LocalLoader loader : loaders) {
518                 clazz = loader.loadClassLocal(className, resolve);
519                 if (clazz != null) {
520                     return clazz;
521                 }
522             }
523         }
524         final LocalLoader fallbackLoader = this.fallbackLoader;
525         if (fallbackLoader != null) {
526             return fallbackLoader.loadClassLocal(className, resolve);
527         }
528         return null;
529     }

531     /**
532      * Load a resource from a local loader.
533      *
534      * @param name the resource name
535      * @return the resource URL, or {@code null} if not found
536      */
537     URL getResource(final String name) {
538         final String canonPath = PathUtils.canonicalize(name);
539         for (String s : Module.systemPaths) {
540             if (canonPath.startsWith(s)) {
541                 return moduleClassLoader.getResource(canonPath);
542             }
543         }
544         log.trace("Attempting to find resource %s in %s", canonPath, this);
545         final String path = pathOf(canonPath);
546         final Map<String, List<LocalLoader>> paths = getPathsUnchecked();
547         final List<LocalLoader> loaders = paths.get(path);
548         if (loaders != null) {
549             for (LocalLoader loader : loaders) {
550                 final List<Resource> resourceList = loader.loadResourceLocal(canonPath);
551                 for (Resource resource : resourceList) {
552                     return resource.getURL();
553                 }
554             }
555         }
556         final LocalLoader fallbackLoader = this.fallbackLoader;
557         if (fallbackLoader != null) {
558             final List<Resource> resourceList = fallbackLoader.loadResourceLocal(canonPath);
559             for (Resource resource : resourceList) {
560                 return resource.getURL();
561             }
562         }
563         return null;
564     }

566     /**
567      * Load all resources of a given name from a local loader.
568      *
569      * @param name the resource name
570      * @return the enumeration of all the matching resource URLs (may be empty)
571      */
572     Enumeration<URL> getResources(final String name) {
573         final String canonPath = PathUtils.canonicalize(name);
574         for (String s : Module.systemPaths) {
575             if (canonPath.startsWith(s)) {
576                 try {
577                     return moduleClassLoader.getResources(canonPath);
578                 } catch (IOException e) {
579                     return ConcurrentClassLoader.EMPTY_ENUMERATION;
580                 }
581             }
582         }
583         log.trace("Attempting to find all resources %s in %s", canonPath, this);
584         final String path = pathOf(canonPath);
585         final Map<String, List<LocalLoader>> paths = getPathsUnchecked();
586         final List<LocalLoader> loaders = paths.get(path);
587 
588         final List<URL> list = new ArrayList<URL>();
589         if (loaders != null) {
590             for (LocalLoader loader : loaders) {
591                 final List<Resource> resourceList = loader.loadResourceLocal(canonPath);
592                 for (Resource resource : resourceList) {
593                     list.add(resource.getURL());
594                 }
595             }
596         }
597         final LocalLoader fallbackLoader = this.fallbackLoader;
598         if (fallbackLoader != null) {
599             final List<Resource> resourceList = fallbackLoader.loadResourceLocal(canonPath);
600             for (Resource resource : resourceList) {
601                 list.add(resource.getURL());
602             }
603         }
604 
605         return list.size() == 0 ? ConcurrentClassLoader.EMPTY_ENUMERATION : Collections.enumeration(list);
606     }

628     /**
629      * Get the path name of a class.
630      *
631      * @param className the binary name of the class
632      * @return the parent path
633      */
634     static String pathOfClass(final String className) {
635         final String resourceName = className.replace('.', '/');
636         final String path;
637         final int idx = resourceName.lastIndexOf('/');
638         if (idx > -1) {
639             path = resourceName.substring(0, idx);
640         } else {
641             path = "";
642         }
643         return path;
644     }

646     /**
647      * Get the path name of a resource.
648      *
649      * @param resourceName the resource name
650      * @return the parent path
651      */
652     static String pathOf(final String resourceName) {
653         final String path;
654         if (resourceName.indexOf('/') == 0) {
655             return pathOf(resourceName.substring(1));
656         }
657         final int idx = resourceName.lastIndexOf('/');
658         if (idx > -1) {
659             path = resourceName.substring(0, idx);
660         } else {
661             path = "";
662         }
663         return path;
664     }

1127         Linkage oldLinkage = this.linkage;
1128         Linkage linkage;
1129         Linkage.State state = oldLinkage.getState();
1130         if (state == Linkage.State.LINKED) {
1131             return oldLinkage.getPaths();
1132         }

1134         boolean intr = false;
1135         try {
1136             for (;;) {
1137                 synchronized (this) {
1138                     oldLinkage = this.linkage;
1139                     state = oldLinkage.getState();
1140                     while (state == Linkage.State.LINKING || state == Linkage.State.NEW) try {
1141                         wait();
1142                         oldLinkage = this.linkage;
1143                         state = oldLinkage.getState();
1144                     } catch (InterruptedException e) {
1145                         intr = true;
1146                     }
1147                     if (state == Linkage.State.LINKED) {
1148                         return oldLinkage.getPaths();
1149                     }
1150                     this.linkage = linkage = new Linkage(oldLinkage.getSourceList(), Linkage.State.LINKING);
1151                     // fall out and link
1152                 }
1153                 boolean ok = false;
1154                 try {
1155                     link(linkage);
1156                     ok = true;
1157                 } finally {
1158                     if (! ok) {
1159                         // restore original (lack of) linkage
1160                         synchronized (this) {
1161                             if (this.linkage == linkage) {
1162                                 this.linkage = oldLinkage;
1163                                 notifyAll();
1164                             }
1165                         }
1166                     }
1167                 }
1168             }
1169         } finally {
1170             if (intr) {
