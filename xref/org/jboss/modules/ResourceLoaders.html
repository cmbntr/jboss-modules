23  package org.jboss.modules;
24  
25  import java.io.File;
26  import java.security.AccessController;
27  import java.util.jar.JarFile;
28  import org.jboss.modules.filter.PathFilter;
29  
30  /**
31   * Static factory methods for various types of resource loaders.
32   *
33   * @apiviz.exclude
34   *
35   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
36   */
37  public final class ResourceLoaders {
38      static final boolean WRITE_INDEXES;
39  
40      static {
41          WRITE_INDEXES = Boolean.parseBoolean(AccessController.doPrivileged(new PropertyReadAction("jboss.modules.write-indexes", "true")));
42      }
43  
44      private ResourceLoaders() {
45      }
46  
47      /**
48       * Create a filesystem-backed resource loader with support for native libraries.  Created classes
49       * have a code source with a {@code file:} URL.
50       *
51       * @param name the name of the resource root
52       * @param root the root file of the resource loader
53       * @return the resource loader
54       */
55      public static ResourceLoader createFileResourceLoader(final String name, final File root) {
56          return new FileResourceLoader(name, root);
57      }
58  
59      /**
60       * Create a JAR-backed resource loader.  JAR resource loaders do not have native library support.
61       * Created classes have a code source with a {@code jar:} URL; nested JARs are not supported.
62       *
63       * @param name the name of the resource root
64       * @param jarFile the backing JAR file
65       * @return the resource loader
66       */
67      public static ResourceLoader createJarResourceLoader(final String name, final JarFile jarFile) {
68          return new JarFileResourceLoader(name, jarFile);
69      }
70  
71      /**
72       * Create a filtered view of a resource loader, which allows classes to be included or excluded on a name basis.
73       * The given filter is matched against the actual class or resource name, not the directory name.
74       *
75       * @param pathFilter the path filter to apply
76       * @param originalLoader the original loader to apply to
77       * @return the filtered resource loader
78       */
79      public static ResourceLoader createFilteredResourceLoader(final PathFilter pathFilter, final ResourceLoader originalLoader) {
80          return new FilteredResourceLoader(pathFilter, originalLoader);
81      }
82  }
