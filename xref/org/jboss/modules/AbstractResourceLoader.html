32   * An abstract resource loader implementation.

34   * @author <a href="mailto:cdewolf@redhat.com">Carlo de Wolf</a>

36  public abstract class AbstractResourceLoader implements ResourceLoader {
37      private static String getDefinedAttribute(Attributes.Name name, Attributes entryAttribute, Attributes mainAttribute) {
38          final String value = entryAttribute == null ? null : entryAttribute.getValue(name);
39          return value == null ? mainAttribute == null ? null : mainAttribute.getValue(name) : value;
40      }

42      /**
43       * Convenience method to get a package specification from a {@code Manifest}.
44       *
45       * @param name the (dot-separated) package name
46       * @param manifest the {@code Manifest} instance
47       * @param rootUrl the code source URL
48       * @return the package specification
49       */
50      protected static PackageSpec getPackageSpec(final String name, final Manifest manifest, final URL rootUrl) {
51          final PackageSpec spec = new PackageSpec();
52          if (manifest == null) {
53              return null;
54          }
55          final Attributes mainAttribute = manifest.getMainAttributes();
56          final String path = name.replace('.', '/').concat("/");
57          final Attributes entryAttribute = manifest.getAttributes(path);
58          spec.setSpecTitle(getDefinedAttribute(Attributes.Name.SPECIFICATION_TITLE, entryAttribute, mainAttribute));
59          spec.setSpecVersion(getDefinedAttribute(Attributes.Name.SPECIFICATION_VERSION, entryAttribute, mainAttribute));
60          spec.setSpecVendor(getDefinedAttribute(Attributes.Name.SPECIFICATION_VENDOR, entryAttribute, mainAttribute));
61          spec.setImplTitle(getDefinedAttribute(Attributes.Name.IMPLEMENTATION_TITLE, entryAttribute, mainAttribute));
62          spec.setImplVersion(getDefinedAttribute(Attributes.Name.IMPLEMENTATION_VERSION, entryAttribute, mainAttribute));
63          spec.setImplVendor(getDefinedAttribute(Attributes.Name.IMPLEMENTATION_VENDOR, entryAttribute, mainAttribute));
64          if (Boolean.parseBoolean(getDefinedAttribute(Attributes.Name.SEALED, entryAttribute, mainAttribute))) {
65              spec.setSealBase(rootUrl);
66          }
67          return spec;
68      }
