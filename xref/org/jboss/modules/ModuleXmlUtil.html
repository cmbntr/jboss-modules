23  package org.jboss.modules;
24  
25  import java.io.File;
26  
27  /**
28   * Utility class for default module file names.
29   * <p/>
30   * Date: 10.05.2011
31   *
32   * @author <a href="mailto:jperkins@redhat.com">James R. Perkins</a>
33   */
34  class ModuleXmlUtil {
35      public static final String DEFAULT_FILENAME = "module.xml";
36  
37      /**
38       * Private constructor.
39       */
40      private ModuleXmlUtil() {
41      }
42  
43      /**
44       * Creates a file based on the directory and the module identifier.
45       * <p/>
46       * The {@code dir} parameter must be a directory and the {@code identifier} cannot be {@code null}.
47       *
48       * @param dir        the base directory where the file should be located.
49       * @param identifier the module identifier.
50       *
51       * @return the module XML file.
52       * @throws IllegalArgumentException if the {@code dir} parameter is not a directory or an argument is {@code null}.
53       */
54      public static File toFile(final File dir, final ModuleIdentifier identifier) {
55          return toFile(dir, DEFAULT_FILENAME, identifier);
56      }
57  
58      /**
59       * Creates a file based on the directory and the module identifier.
60       * <p/>
61       * The {@code dir} parameter must be a directory and the {@code identifier} cannot be {@code null}.
62       *
63       * @param dir        the base directory where the file should be located.
64       * @param name       the name of the XML file.
65       * @param identifier the module identifier.
66       *
67       * @return the module XML file.
68       * @throws IllegalArgumentException if the {@code dir} parameter is not a directory or an argument is {@code null}.
69       */
70      public static File toFile(final File dir, final String name, final ModuleIdentifier identifier) {
71          if (dir == null || !dir.isDirectory()) {
72              throw new IllegalArgumentException(String.format("Must be a directory. File %s is not a directory.", dir));
73          }
74          if (name == null || name.isEmpty()) {
75              throw new IllegalArgumentException("The name cannot be empty.");
76          }
77          if (identifier == null) {
78              throw new IllegalArgumentException("The module identifier cannot be null.");
79          }
80          return new File(dir, baseFilename(name, identifier));
81      }
82  
83      /**
84       * Creates a path name from the module identifier. The name always ends with the separator character.
85       * </p>
86       * A {@code null} identifier will result in no separator being used.
87       *
88       * @param identifier the module identifier.
89       * @param separator  the directory separator.
90       *
91       * @return a path name of the module identifier.
92       * @throws IllegalArgumentException if the module identifier is {@code null}.
93       */
94      public static String baseDirectory(final ModuleIdentifier identifier, final String separator) {
95          if (identifier == null) {
96              throw new IllegalArgumentException("The module identifier cannot be null.");
97          }
98          final String namePath = identifier.getName().replace('.', File.separatorChar);
99          final StringBuilder baseName = new StringBuilder();
100         baseName.append(namePath).
101                 append((separator == null ? "" : separator)).
102                 append(identifier.getSlot()).
103                 append((separator == null ? "" : separator));
104         return baseName.toString();
105     }
106 
107     /**
108      * Creates a path name from the module identifier with the default {@link java.io.File#separator} character.
109      *
110      * @param identifier the module identifier.
111      *
112      * @return a path name of the module identifier.
113      * @throws IllegalArgumentException if the module identifier is {@code null}.
114      * @see #baseDirectory(ModuleIdentifier, String)
115      */
116     public static String baseDirectory(final ModuleIdentifier identifier) {
117         return baseDirectory(identifier, File.separator);
118     }
119 
120     /**
121      * Creates a path name to the module XML file from the module identifier. Uses the {@link #DEFAULT_FILENAME} for
122      * the XML file name.
123      *
124      * @param identifier the module identifier.
125      *
126      * @return a path name to the module XML file.
127      * @throws IllegalArgumentException if the module identifier is {@code null}.
128      */
129     public static String baseFilename(final ModuleIdentifier identifier) {
130         return baseFilename(DEFAULT_FILENAME, identifier);
131     }
132 
133     /**
134      * Creates a path name to the module XML file from the module identifier.
135      *
136      * @param name       the XML file name.
137      * @param identifier the module identifier.
138      *
139      * @return a path name to the module XML file.
140      * @throws IllegalArgumentException if the module identifier is {@code null}.
141      */
142     public static String baseFilename(final String name, final ModuleIdentifier identifier) {
143         return baseDirectory(identifier) + name;
144     }
145 
146     /**
147      * Creates a path name to the module XML
148      *
149      * @param name       the XML file name. file from the module identifier.
150      * @param separator  the directory separator.
151      * @param identifier the module identifier.
152      *
153      * @return a path name to the module XML file.
154      * @throws IllegalArgumentException if the module identifier is {@code null}.
155      */
156     public static String baseFilename(final String name, final String separator, final ModuleIdentifier identifier) {
157         return baseDirectory(identifier, separator) + name;
158     }
