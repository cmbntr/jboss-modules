23  package org.jboss.modules.ref.util;
24  
25  import static org.junit.Assert.assertEquals;
26  import static org.junit.Assert.assertNotNull;
27  import static org.junit.Assert.assertSame;
28  import static org.junit.Assert.fail;
29  
30  import org.jboss.modules.ref.Reaper;
31  import org.jboss.modules.ref.Reference;
32  import org.jboss.modules.ref.Reference.Type;
33  
34  /**
35   * Assertion methods used by reference tests.
36   * 
37   * @author <a href="mailto:flavia.rainone@jboss.com">Flavia Rainone</a>
38   */
39  public class Assert {
40  
41      /**
42       * Assert that {@code reference} contains {@code referent}, {@code attachment}, and that its type is {@code type}.
43       * 
44       * @param <T>         the type of referent
45       * @param <A>         the type of attachment
46       * @param reference   the reference that needs to be checked
47       * @param referent    the expected content for {@code reference}
48       * @param attachment  the expected attachment contained in {@code reference}
49       * @param type        the expected type of {@code reference}
50       */
51      public static final <T, A> void assertReference(Reference<T, A> reference, T referent, A attachment, Type type) {
52          assertEquals(referent, reference.get());
53          assertEquals(attachment, reference.getAttachment());
54          assertSame(type, reference.getType());
55          assertNotNull(reference.toString()); // make sure that no exception is thrown and that null is never returned
56      }
57  
58      /**
59       * Assert that {@code reference} contains {@code referent}, {@code attachment}, and that its type is {@code type}.
60       * Also asserts that {@code reaper} is the reaper used by {@code referent}.
61       * <p>
62       * Call this method only for {@link org.jboss.msc.ref.Reapable Reapable} references. This method assumes that 
63       * {@code reference} has a method called {@code getReaper} that will return the reaper.
64       * 
65       * @param <T>         the type of referent
66       * @param <A>         the type of attachment
67       * @param reference   the {@code Reapable} reference that needs to be checked
68       * @param referent    the expected content for {@code reference}
69       * @param attachment  the expected attachment contained in {@code reference}
70       * @param type        the expected type of {@code reference}
71       * @param reaper      the expected reaper
72       */
73      public static final <T, A> void assertReference(Reference<T, A> reference, T referent, A attachment, Type type, Reaper<T, A> reaper) {
74          assertReference(reference, referent, attachment, type);
75          try {
76              assertEquals(reaper, reference.getClass().getMethod("getReaper").invoke(reference));
77          } catch (NoSuchMethodException e) {
78              fail("Can't find getReaper method at clas " + reference.getClass());
79          } catch (RuntimeException e) {
80              throw e;
81          }
82          catch (Exception e) {
83              fail(e.getMessage());
84          }
85      }
86  }
