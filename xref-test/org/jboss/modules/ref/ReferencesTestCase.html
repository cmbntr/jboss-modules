161     private Reference<String, String> createReference(Type type) {
162         final Reference<String, String> reference = References.create(type, "value", "attachment");
163         assertReference(reference, "value", "attachment", type);
164         return reference;
165     }

167     private Reference<Object, String> createReferenceWithReaper(Type type, boolean expectedNullValue, boolean testReaper) throws Exception {
168         Object referent = new Object();
169         final TestReaper<Object, String> reaper = new TestReaper<Object, String>();
170         final Reference<Object, String> reference = References.create(type, referent, "attachment", reaper);
171         assertReference(reference, expectedNullValue? null: referent, "attachment", type, reaper);
172         if (testReaper) {
173             referent = null;
174             System.gc();
175             assertSame(reference, reaper.get());
176         }
177         return reference;
178     }

180     private Reference<Collection<Integer>, Object> createReferenceWithQueue(Type type, boolean expectedNullValue, boolean testQueue) throws Exception {
181         final Object referenceAttachment = new Object();
182         Collection<Integer> referent = new ArrayList<Integer>();
183         referent.add(Integer.valueOf(1));
184         referent.add(Integer.valueOf(11));
185         referent.add(Integer.valueOf(111));
186         final ReferenceQueue<Collection<Integer>> referenceQueue = new ReferenceQueue<Collection<Integer>>();
187         final Reference<Collection<Integer>, Object> reference = References.create(type, referent, referenceAttachment, referenceQueue);
188         assertReference(reference, expectedNullValue? null: referent, referenceAttachment, type, null);
189         if (testQueue) {
190             referent = null;
191             System.gc();
192             assertSame(reference, referenceQueue.remove(300));
193         }
194         return reference;
195     }
