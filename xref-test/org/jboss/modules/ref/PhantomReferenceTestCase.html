37  /**
38   * Test for PhantomReference.
39   * 
40   * @author <a href="mailto:flavia.rainone@jboss.com">Flavia Rainone</a>
41   * @see PhantomReference
42   */
43  public class PhantomReferenceTestCase extends AbstractReapableReferenceTest {
44  
45      @Test
46      public void plainPhantomReference() {
47          final Reference<String, String> reference = new PhantomReference<String, String>("referent", "attachment",
48                  (ReferenceQueue<String>) null);
49          assertReference(reference, null, "attachment", null);
50      }
51  
52      @Test
53      public void nullPhantomReference() throws Exception {
54          final Reference<Thing, Integer> reference = new PhantomReference<Thing, Integer>(null, Integer.valueOf(0), new ReferenceQueue<Thing>());
55          assertReference(reference, null, Integer.valueOf(0), null);
56      }
57  
58      @Test
59      public void phantomReferenceWithReferenceQueue() throws Exception {
60          final ReferenceQueue<Collection<Object>> referenceQueue = new ReferenceQueue<Collection<Object>>();
61          Collection<Object> collection = new ArrayList<Object>();
62          final Reference<Collection<Object>, String> reference = new PhantomReference<Collection<Object>, String>(collection, "collection", referenceQueue);
63          assertReference(reference, null, "collection", null);
64          collection = null;
65          System.gc();
66          assertSame(reference, referenceQueue.remove(300));
67      }
68  
69      @Test
70      public void phantomReferenceWithReaper() throws Exception {
71          Thing thing = new Thing();
72          final TestReaper<Thing, Void> reaper = new TestReaper<Thing, Void>();
73          final Reference<Thing, Void> reference = new PhantomReference<Thing, Void>(thing, null, reaper);
74          assertReference(reference, null, null, reaper);
75          thing = null;
76          System.gc();
77          assertSame(reference, reaper.getReapedReference());
78      }
79  
80      @Override @Test
81      public void clearReference() {
82          final Reference<Boolean, String> reference = createReference(Boolean.TRUE, "attachment for true");
83          assertNull(reference.get());
84          assertEquals("attachment for true", reference.getAttachment());
85  
86          reference.clear();
87          assertNull(reference.get());
88          assertEquals("attachment for true", reference.getAttachment());
89      }
90  
91      @Override
92      <T, A> Reference<T, A> createReference(T value, A attachment) {
93          return new PhantomReference<T, A>(value, attachment, new TestReaper<T, A>());
94      }
95  
96      @Override
97      Type getTestedReferenceType() {
98          return Type.PHANTOM;
99      }
