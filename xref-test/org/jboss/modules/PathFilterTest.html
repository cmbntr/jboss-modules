23  package org.jboss.modules;

25  import org.jboss.modules.filter.MultiplePathFilterBuilder;
26  import org.jboss.modules.filter.PathFilter;
27  import org.jboss.modules.filter.PathFilters;
28  import org.junit.Test;
29  
30  import static org.junit.Assert.assertFalse;
31  import static org.junit.Assert.assertTrue;
32  
33  /**
34   * Test to verify the PathFilter functionality.
35   * 
36   * @author John E. Bailey
37   */
38  public class PathFilterTest {
39  
40      @Test
41      public void testMatch() throws Exception {
42          PathFilter pathFilter = PathFilters.match("foo/**");
43          assertFalse(pathFilter.accept("foo"));
44          assertTrue(pathFilter.accept("foo/bar"));
45          assertTrue(pathFilter.accept("foo/bar/baz"));
46  
47          pathFilter = PathFilters.match("foo/*");
48          assertFalse(pathFilter.accept("foo"));
49          assertTrue(pathFilter.accept("foo/bar"));
50          assertTrue(pathFilter.accept("foo/bar/baz"));
51  
52          pathFilter = PathFilters.match("foo");
53          assertTrue(pathFilter.accept("foo"));
54          assertTrue(pathFilter.accept("foo/bar"));
55          assertTrue(pathFilter.accept("foo/bar/baz"));
56  
57          pathFilter = PathFilters.match("**/bar/**");
58          assertFalse(pathFilter.accept("foo"));
59          assertFalse(pathFilter.accept("foo/bar"));
60          assertTrue(pathFilter.accept("foo/bar/baz"));
61          assertTrue(pathFilter.accept("foo/baz/bar/biff"));
62      }
63  
64      @Test
65      public void testDelegating() throws Exception {
66          final MultiplePathFilterBuilder builder = PathFilters.multiplePathFilterBuilder(true);
67          builder.addFilter(PathFilters.match("foo/*"), false);
68          builder.addFilter(PathFilters.match("**/bar/**"), false);
69          builder.addFilter(PathFilters.match("baz/**"), false);
70          PathFilter pathFilter = builder.create();
71          assertTrue(pathFilter.accept("foo"));
72          assertFalse(pathFilter.accept("foo/bar"));
73          assertFalse(pathFilter.accept("foo/bar/baz"));
74          assertFalse(pathFilter.accept("baz/foo/bar"));
75      }
76  }
