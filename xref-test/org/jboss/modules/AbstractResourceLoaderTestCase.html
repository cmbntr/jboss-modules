42  /**
43   * Abstract Test Case used as the base for all resource loader tests.
44   *
45   * @author John Bailey
46   */
47  public abstract class AbstractResourceLoaderTestCase extends AbstractModuleTestCase {
48  
49      protected ResourceLoader loader;
50  
51      @Before
52      public void setupLoader() throws Exception {
53          loader = createLoader(PathFilters.acceptAll());
54      }
55  
56      protected abstract ResourceLoader createLoader(final PathFilter exportFilter) throws Exception;
57      protected abstract void assertResource(final Resource resource, final String fileName);
58  
59      @Test
60      public void testBasicResource() throws Exception {
61          Resource resource = loader.getResource("/test.txt");
62          assertNotNull(resource);
63          assertResource(resource, "test.txt");
64          resource = loader.getResource("/nested/nested.txt");
65          assertNotNull(resource);
66          assertResource(resource, "nested/nested.txt");
67      }
68  
69      @Test
70      public void testMissingResource() throws Exception {
71          Resource resource = loader.getResource("/test-bogus.txt");
72          assertNull(resource);
73      }
74  
75      @Test
76      public void testIndexPaths() throws Exception {
77          final Collection<String> paths = loader.getPaths();
78          assertFalse(paths.isEmpty());
79  
80          assertTrue(paths.contains(""));
81          assertTrue(paths.contains("META-INF"));
82          assertTrue(paths.contains("nested"));
83          assertTrue(paths.contains("org"));
84          assertTrue(paths.contains("org/jboss"));
85          assertTrue(paths.contains("org/jboss/modules"));
86          assertTrue(paths.contains("org/jboss/modules/test"));
87      }
88  
89      @Test
90      public void testGetClassSpec() throws Exception {
91          ClassSpec spec = loader.getClassSpec(Module.fileNameOfClass("org.jboss.modules.test.TestClass"));
92          assertNotNull(spec);
93          byte[] bytes = spec.getBytes();
94  
95          final URL classResource = getClass().getClassLoader().getResource("org/jboss/modules/test/TestClass.class");
96          final byte[] expectedBytes = readBytes(classResource.openStream());
97          assertArrayEquals(expectedBytes, bytes);
98      }
99  
100     @Test
101     public void testMissingClassSpec() throws Exception {
102         ClassSpec spec = loader.getClassSpec(Module.fileNameOfClass("org.jboss.modules.test.BogusClass"));
103         assertNull(spec);
104     }
105 
106     @Test
107     public void testGetPackageSpec() throws Exception {
108         PackageSpec spec = loader.getPackageSpec("org.jboss.modules.test");
109         assertNotNull(spec);
110 
111         assertEquals("JBoss Modules Test Classes", spec.getSpecTitle());
112         assertEquals("0.1", spec.getSpecVersion());
113         assertEquals("JBoss", spec.getSpecVendor());
114         assertEquals("org.jboss.modules.test", spec.getImplTitle());
115         assertEquals("1.0", spec.getImplVersion());
116         assertEquals("JBoss", spec.getImplVendor());
117     }
118 
119     @Test
120     public void testMissingPackageSpec() throws Exception {
121         PackageSpec spec = loader.getPackageSpec("org.jboss.modules.bogus");
122         assertNotNull(spec);
123 
124         assertEquals("MODULES-89", spec.getSpecTitle());
125         assertNull(spec.getSpecVersion());
126         assertNull(spec.getSpecVendor());
127         assertNull(spec.getImplTitle());
128         assertNull(spec.getImplVersion());
129         assertNull(spec.getImplVendor());
130     }
